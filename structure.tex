%% Command: cvheader
%% Parameters:
%%      #1 => Image;
%%      #2 => Additional information;
\newcommand{\cvuserinfo}[2]{
     \begin{center}
          % Cropping the image to be a circle instead of a square
          \begin{tikzpicture}
               \clip (0,0) circle (1.7cm) node {
                    \includegraphics[width=2.5cm]{#1}
               };
          \end{tikzpicture}\\[10pt]
          % Adding the rest of the content
          #2
     \end{center}
}

%% Command: cvsection
%% Description:
%%      Draws a section much like the following one
%%         SECTION TITLE
%%      -------------------------------------------
%%      Some really interesting content.......
%% Parameters:
%%      #1 => Section title;
%%      #2 => Section content;
\newcommand{\cvsection}[2]{
     \begin{minipage}{\linewidth}
          \textcolor{foreground00}{\huge{\robotoserifBold{#1}}}
          \vspace{5pt}
          \color{background00}\hrule height 2pt
          \vspace{10pt}
          \textcolor{foreground02}{\robotoserifRegular{#2}}
     \end{minipage}\\[10pt]
}

\newcommand{\cvpersonalcontact}[2]{%
     \begin{minipage}{10pt}
          \textcolor{foreground01}{\faIcon{#1}}
     \end{minipage}
     \textcolor{foreground02}{\robotoserifRegular{#2}}\\[4pt]
}

\newcommand{\cvtitle}[3]{
     \begin{minipage}[t][2.5cm][t]{\textwidth}
          \begin{vplace}
               \uppercase{\HUGE\robotoserifBold{#1}}\\[5pt]
               \color{foreground01}\large\robotoserifMedium{#2}\\
          \end{vplace}
     \end{minipage}\\
     \begin{minipage}[t][.75cm][b]{\textwidth}
          #3
     \end{minipage}
}

\newcommand{\cvexperience}[5]{%
     \begin{minipage}[t][][l]{0.7\textwidth}
          \robotoserifMedium{\Large\textcolor{black}{#1}}
     \end{minipage}\\[2pt]
     \textcolor{foreground01}{\robotoserifRegular{#2}}\\[5pt]
     \begin{minipage}[t][10pt][t]{0.5\textwidth}
          \hspace{10pt}\cvpersonalcontact{calendar}{#3}
     \end{minipage}
     \begin{minipage}[t][10pt][t]{0.5\textwidth}
          \cvpersonalcontact{map-marker}{#4}
     \end{minipage}\\[5pt]
     #5
}
