%% Command: cvheader
%% Parameters:
%%      #1 => Image;
%%      #2 => Additional information;
\newcommand{\cvuserinfo}[2]{
     \begin{center}
          % Cropping the image to be a circle instead of a square
          \begin{tikzpicture}
               \clip (0,0) circle (1.7cm) node {
                    \includegraphics[width=2.5cm]{#1}
               };
          \end{tikzpicture}\\[10pt]
          % Adding the rest of the content
          #2
     \end{center}
}

%% Command: cvsection
%% Description:
%%      Draws a section much like the following one
%%         SECTION TITLE
%%      -------------------------------------------
%%      Some really interesting content.......
%% Parameters:
%%      #1 => Section title;
%%      #2 => Section content;
\newcommand{\cvsection}[2]{
     \begin{minipage}{\linewidth}
          \textcolor{foreground00}{\Huge{\robotoBold{#1}}}
          \vspace{5pt}
          \color{background00}\hrule height 2pt
          \vspace{10pt}
          \color{foreground02}\robotoMedium{#2}
     \end{minipage}
}

\newcommand{\cvpersonalcontact}[2]{
     \begin{minipage}{10pt}
          {\color{foreground01}\faIcon{#1}}
     \end{minipage}
     {\color{foreground02}\robotoMedium{#2}}\\[4pt]
}

\newcommand{\cvtitle}[3]{
     \begin{minipage}[t][2.5cm][t]{\textwidth}
          \begin{vplace}
               \uppercase{\HUGE\robotoBold{#1}}\\[5pt]
               \color{foreground01}\large\robotoMedium{#2}\\
          \end{vplace}
     \end{minipage}\\
     \begin{minipage}[t][.75cm][b]{\textwidth}
          #3
     \end{minipage}
}
